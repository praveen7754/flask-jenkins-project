pipeline {
    agent any

    stages {

        stage('Checkout') {
            steps {
                checkout scm
            }
        }

        stage('Python Version') {
            steps {
                bat 'py --version'
            }
        }

        stage('Install Dependencies') {
            steps {
                bat 'py -m pip install -r requirements.txt'
            }
        }

        stage('Test') {
            steps {
                bat 'py -m py_compile app.py'
            }
        }

        stage('Deploy') {
            steps {
                bat '''
                echo Deploying Application...
                '''
            }
        }
    }
}
